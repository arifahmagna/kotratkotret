// Inisialisasi array untuk menyimpan metode pembayaran
$payment_methods = [];

try {
    // Query untuk mengambil metode pembayaran
    $sql = "SELECT id_tmpm, paymentMethod_tmpm FROM tbl_m_paymentmethod";
    $result = $conn->query($sql);

    // Mengecek apakah query mengembalikan hasil
    if ($result->num_rows > 0) {
        // Ambil semua hasil sebagai array asosiatif
        while ($row = $result->fetch_assoc()) {
            $payment_methods[] = $row;
        }
    } else {
        echo "Tidak ada metode pembayaran yang ditemukan.";
    }
} catch (Exception $e) {
    // Menangani error umum
    echo "Error: " . $e->getMessage();
